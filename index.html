P7914 [CSP-S 2021] 括号序列 题解

创建于2022-08-07

其实T2想清楚就不是很难，(虽然想清楚也不简单)

我这里分享一种很自然的想法，当然是区间dp啦

区间dp分6种状态

  1. ***的种类数，这种情况相当与题目中的SSS，2到5中都一样

  2. (...)的种类数，这种情况表示有括号包裹的合法序列，2到5中都一样

  3. (...)***(...)***(...)***的种类数，表示以(...)开头，以***结尾的一长串，没有个数限制，比如(...)***也可以

  4. (...)***(...)***(...)的种类数，表示以(...)开头，以(...)结尾的一长串，没有个数限制，比如(...)也可以

  5. ***(...)***(...)***(...)的种类数，表示以***开头，以(...)结尾的一长串，没有个数限制，比如***(...)也可以

  6. ***(...)***(...)***的种类数，表示以***开头，以***结尾的一长串，没有个数限制，比如***也可以

转移很自然：

dp[0][l][r]=(r−l+1≤k)∗dp[0][l][r−1]∗(s[r]==′∗′∣∣s[r]==′?′)dp[1][l][r]=pipei(l,r)∗(dp[2][l+1][r−1]+dp[3][l+1][r−1]+dp[4][l+1][r−1])dp[0][l][r]=(r-l+1\le k)*dp[0][l][r-1]*(s[r]=='*'||s[r]=='?') \\ dp[1][l][r]=pipei(l,r)*(dp[2][l+1][r-1]+dp[3][l+1][r-1]+dp[4][l+1][r-1]) dp[0][l][r]=(r−l+1≤k)∗dp[0][l][r−1]∗(s[r]==′∗′∣∣s[r]==′?′)dp[1][l][r]=pipei(l,r)∗(dp[2][l+1][r−1]+dp[3][l+1][r−1]+dp[4][l+1][r−1])

其中pipei(l,r)pipei(l,r)pipei(l,r)表示s[l]s[l]s[l]和s[r]s[r]s[r]可以匹配成括号

dp[2][l][r]=∑dp[3][l][i]∗dp[0][i+1][r]dp[3][l][r]=∑(dp[2][l][i]+dp[3][l][i])∗dp[1][i+1][r]dp[4][l][r]=∑(dp[5][l][i]+dp[4][l][i])∗dp[1][i+1][r]dp[5][l][r]=∑dp[4][l][i]∗dp[0][i+1][r]dp[2][l][r]=\sum dp[3][l][i]*dp[0][i+1][r] \\ dp[3][l][r]=\sum (dp[2][l][i]+dp[3][l][i])*dp[1][i+1][r] \\ dp[4][l][r]=\sum (dp[5][l][i]+dp[4][l][i])*dp[1][i+1][r] \\ dp[5][l][r]=\sum dp[4][l][i]*dp[0][i+1][r] dp[2][l][r]=∑dp[3][l][i]∗dp[0][i+1][r]dp[3][l][r]=∑(dp[2][l][i]+dp[3][l][i])∗dp[1][i+1][r]dp[4][l][r]=∑(dp[5][l][i]+dp[4][l][i])∗dp[1][i+1][r]dp[5][l][r]=∑dp[4][l][i]∗dp[0][i+1][r]

最后不要忘了把dp[l][r][0]dp[l][r][0]dp[l][r][0]算到dp[l][r][5]dp[l][r][5]dp[l][r][5]里，还有dp[l][r][1]dp[l][r][1]dp[l][r][1]算到dp[l][r][3]dp[l][r][3]dp[l][r][3]里，不知到为什么可以重新看一遍dp六种状态的定义

我的考场代码（不要学我#define int long long啊）：

#include<bits/stdc++.h>
#define int long long
using namespace std;
const int N=510,mod=1e9+7;
int n,k,dp[6][N][N];
char s[N];
bool pipei(int l,int r){
  return (s[l]=='('||s[l]=='?')&(s[r]==')'||s[r]=='?');
}
signed main(){
  freopen("bracket.in","r",stdin);
  freopen("bracket.out","w",stdout);
  ios::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);
  cin>>n>>k>>s+1;
  for(int i=1;i<=n;i++)dp[0][i][i-1]=1;
  for(int len=1;len<=n;len++){
    for(int l=1;l+len-1<=n;l++){
      int r=l+len-1;
      if(len<=k)dp[0][l][r]=dp[0][l][r-1]&&(s[r]=='*'||s[r]=='?');
      if(len>=2){
        dp[1][l][r]=pipei(l,r)*(dp[2][l+1][r-1]+dp[3][l+1][r-1]+dp[4][l+1][r-1]+dp[0][l+1][r-1])%mod;
        for(int i=l;i<r;i++){
          (dp[2][l][r]+=dp[3][l][i]*dp[0][i+1][r])%=mod;
          (dp[3][l][r]+=(dp[2][l][i]+dp[3][l][i])*dp[1][i+1][r])%=mod;
          (dp[4][l][r]+=(dp[5][l][i]+dp[4][l][i])*dp[1][i+1][r])%=mod;
          (dp[5][l][r]+=dp[4][l][i]*dp[0][i+1][r])%=mod;
        }
      }
      (dp[5][l][r]+=dp[0][l][r])%=mod;
      (dp[3][l][r]+=dp[1][l][r])%=mod;
    }
  }
  cout<<dp[3][1][n];
}
